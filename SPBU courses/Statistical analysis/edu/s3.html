for i, n in enumerate(sample_sizes):
    #  
    data = np.random.normal(mu, sigma, size=n)

    #    
    x_vals = np.linspace(min(data) - 1, max(data) + 1, 1000)
    ecdf = np.array([np.sum(data <= x) / n for x in x_vals])

    #   
    true_cdf = stats.norm.cdf(x_vals, loc=mu, scale=sigma)

    #     
    mu_hat, sigma_hat = np.mean(data), np.std(data, ddof=1)
    estimated_cdf = stats.norm.cdf(x_vals, loc=mu_hat, scale=sigma_hat)

    #   
    ax = axes[i // 2, i % 2]

    #  
    ax.plot(x_vals, ecdf, label="  (F_n)", linestyle="dashed")
    ax.plot(x_vals, true_cdf, label="  (F)", color="black")
    ax.plot(x_vals, estimated_cdf, label="   (F_)", linestyle="dotted")

    #  
    ax.set_title(f"n = {n}")
    ax.set_xlabel("x")
    ax.set_ylabel("F(x)")
    ax.legend()
